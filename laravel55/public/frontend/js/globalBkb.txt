	/*fixed header script*/
var stickyOffset = $('.head').offset().top+1;
$(window).scroll(function(){
  var sticky = $('.head'),
      scroll = $(window).scrollTop();
  if (scroll >= stickyOffset) sticky.addClass('sticky');
  else sticky.removeClass('sticky');
});
